#!/bin/bash
TODAY=`date +%Y%m%d`
mkdir backup-$TODAY

for FILE in `ls -R *.txt` 
do
   cp $FILE backup-$TODAY
done

# checks if the day is a Sunday:
DAY=`date +%a`
if [ $DAY = 'Sun' ]
then
Ê  echo 'Backup day'
fi
